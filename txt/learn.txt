内容块属性CSS
1.引入顺序（必须严格遵守）
<!-- 1. Tailwind CSS -->
<script src="https://cdn.tailwindcss.com"></script>

<!-- 2. Font Awesome 图标 -->
<link href="https://cdn.jsdelivr.net/npm/font-awesome@4.7.0/css/font-awesome.min.css" rel="stylesheet">

<!-- 3. 内容块核心CSS -->
<link rel="stylesheet" href="path/to/content-block.css">

2.Tailwind 配置（必需）
<script>
  tailwind.config = {
    theme: {
      extend: {
        colors: {
          primary: '#165DFF',    // 主色
          secondary: '#722ED1',  // 辅助色
          success: '#00B42A',    // 成功色
          warning: '#FF7D00',    // 警告色
          danger: '#F53F3F',     // 危险色
          info: '#86909C',       // 信息色
          dark: '#1D2129',       // 深色
          light: '#F2F3F5'       // 浅色
        },
        fontFamily: {
          inter: ['Inter', 'system-ui', 'sans-serif'],
        },
        boxShadow: {
          'card': '0 4px 20px 0 rgba(0, 0, 0, 0.05)', // 卡片默认阴影
          'card-hover': '0 10px 30px 0 rgba(22, 93, 255, 0.15)', // 卡片悬停阴影
        }
      },
    }
  }
</script>

3.核心使用
content-block	内容块基础类（必加）
content-block-hover	悬停上浮 + 阴影增强效果
content-block-gradient	渐变背景（需配合 Tailwind 渐变类）
content-block-gradient-border	渐变边框效果（需嵌套子元素）
content-block-loading	加载状态（脉冲动画）
text-balance	文本自动平衡排版

4.示例用法
<div class="content-block bg-white p-6 shadow-card content-block-transition content-block-hover">
  <div class="w-12 h-12 bg-primary/10 rounded-lg flex items-center justify-center mb-4">
    <i class="fa fa-file-text-o text-primary text-xl"></i>
  </div>
  <h3 class="text-lg font-semibold text-dark mb-2">示例内容块</h3>

  </a>
</div>


//表格制造==================================================================================================
<table>	表格容器（根标签）	所有表格内容必须包裹其中
<tr>	表格行（table row）	代表表格的一行
<td>	普通单元格（table data）	行内的普通数据单元格
<th>	表头单元格（table header）	行内的标题单元格（默认加粗居中）

colspan="2" 横向合并2行
rowspan="2"纵向合并2行

<table 
  border="1"        <!-- 边框宽度（0=隐藏边框） -->
  width="500"       <!-- 表格宽度（像素px或百分比%） -->
  height="200"      <!-- 表格高度 -->
  align="center"    <!-- 表格整体对齐方式（center/left/right） -->
  cellpadding="10"  <!-- 单元格内边距（内容与边框的距离） -->
  cellspacing="0"   <!-- 单元格间距（边框之间的距离，0=合并边框） -->
>
  <!-- 行和单元格 -->
</table>

<!-- 基础表格：3列2行（1行表头 + 1行数据） -->
                    <table border="1">
                      <!-- 表头行 -->
                      <tr>
                        <th>姓名</th>
                        <th>年龄</th>
                        <th>城市</th>
                      </tr>
                      <!-- 数据行1 -->
                      <tr>
                        <td>张三</td>
                        <td>25</td>
                        <td>北京</td>
                      </tr>
                      <!-- 数据行2 -->
                      <tr>
                        <td>李四</td>
                        <td>30</td>
                        <td>上海</td>
                      </tr>
                    </table>
